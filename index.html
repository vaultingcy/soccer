// 카카오톡 공유 공식 연동 및 연결 링크 발송 함수 function sendMessage(text, isResult = false) { // ⚠️ 중요: 카카오 개발자 센터에서 발급받은 본인의 '자바스크립트 키'를 꼭 입력해야 작동합니다. // 키가 정상 등록되어 있고, 카카오 SDK가 연결되었을 때 실행됩니다. if (typeof Kakao !== 'undefined' && Kakao.isInitialized()) { // 현재 이 웹페이지의 주소를 자동으로 가져옵니다. // (나중에 서버나 Github Pages 등에 올렸을 때 해당 주소로 친구들이 접속하게 됩니다.) const currentUrl = window.location.href; Kakao.Share.sendDefault({ objectType: 'text', text: text, link: { mobileWebUrl: currentUrl, webUrl: currentUrl, }, buttons: [ { title: isResult ? '정답 확인 및 점수 보기' : '나도 스코어 예측하기', link: { mobileWebUrl: currentUrl, webUrl: currentUrl, }, }, ], }); } else { // PC 브라우저이거나 카카오 키 설정이 안 되었을 때의 기존 폴백(Fallback) 로직 if (navigator.share) { navigator.share({ title: '월드컵 스코어 게임', text: text + `\n\n링크: ${window.location.href}` }) .catch(console.error); } else { navigator.clipboard.writeText(text + `\n\n링크: ${window.location.href}`); alert("텍스트와 링크가 클립보드에 복사되었습니다! 단톡방에 붙여넣기(Ctrl+V) 하세요."); } } } // 상단 sharePrediction() 함수와 shareFinalResult() 내부의 sendMessage 호출부도 아래처럼 살짝 변경해줍니다. function sharePrediction() { // ... (기존 변수 정의 코드 동일) ... const message = `⚽ [월드컵 스코어 예측 제출]\n\n👤 참여자: ${name}\n⚔️ 경기: ${data.home} vs ${data.away}\n🔮 나의 예측: ${homeS} : ${awayS}\n\n과연 정답을 맞출 수 있을까요?`; sendMessage(message, false); // false를 넘겨 '나도 예측하기' 버튼 활성화 } function shareFinalResult() { // ... (기존 변수 정의 코드 동일) ... const message = `📢 [월드컵 경기 결과 공지]\n\n⚔️ 대상 경기: ${data.home} vs ${data.away}\n정답 스코어 👉 [ ${homeS} : ${awayS} ]\n\n다들 점수 계산해보세요!`; sendMessage(message, true); // true를 넘겨 '정답 확인' 버튼 활성화 }
