// 경기 결과 정산 및 알림 함수 async function settleMatchAndNotify(matchId, finalHomeScore, finalAwayScore) { // 1. 트랜잭션 시작 (DB 연결 객체 가정) const connection = await db.getConnection(); await connection.beginTransaction(); try { // 유저의 현재 점수(total_score)를 함께 가져오기 위해 JOIN 쿼리 사용 const queryStr = ` SELECT p.*, u.user_phone, u.user_name, u.total_score AS current_total FROM predictions p JOIN users u ON p.user_id = u.id WHERE p.match_id = ? `; const [predictions] = await connection.query(queryStr, [matchId]); if (predictions.length === 0) { await connection.commit(); return; } const finalSign = Math.sign(finalHomeScore - finalAwayScore); const alarmList = []; for (let pred of predictions) { let scoreEarned = 0; let resultMessage = ""; // 2. 점수 계산 로직 if (pred.home_score === finalHomeScore && pred.away_score === finalAwayScore) { scoreEarned = 3; // 스코어 정확히 맞춤 resultMessage = `[적중] 스코어(${finalHomeScore}:${finalAwayScore})를 정확히 맞추어 3점을 획득했습니다!`; } else if (Math.sign(pred.home_score - pred.away_score) === finalSign) { scoreEarned = 1; // 승무패만 맞춤 resultMessage = `[절반의 성공] 승패를 맞추어 1점을 획득했습니다. (결과 ${finalHomeScore}:${finalAwayScore})`; } else { scoreEarned = 0; // 틀림 resultMessage = `[아쉽습니다] 예측에 실패했습니다. (결과 ${finalHomeScore}:${finalAwayScore})`; } // 3. 점수 변동이 있을 때만 DB 업데이트 수행 (트랜잭션 내에서 안전하게 차례대로 반영) if (scoreEarned > 0) { await connection.query( "UPDATE users SET total_score = total_score + ? WHERE id = ?", [scoreEarned, pred.user_id] ); } // 4. 알림톡 발송 데이터 수집 (외부 API 호출은 DB 트랜잭션이 성공한 뒤에 보냅니다) alarmList.push({ phoneNumber: pred.user_phone, templateCode: "MATCH_RESULT_ALERT", parameters: { "userName": pred.user_name, "matchName": pred.match_name, "result": resultMessage, "totalScore": Number(pred.current_total) + scoreEarned } }); } // 5. DB 반영 확정 (모든 유저 점수 수정이 성공했을 때만 적용) await connection.commit(); // 6. 비동기로 알림톡 발송 (DB 처리가 끝난 후 병렬 처리하여 대기 시간 축소) Promise.all( alarmList.map(alarmData => sendKakaoAlimtalk(alarmData).catch(err => console.error(`[알림톡 발송 실패] 유저: ${alarmData.parameters.userName}, 사유: ${err.message}`) ) ) ); console.log(`경기 ${matchId} 정산 완료 및 알림 발송 시작`); } catch (error) { // 오류 발생 시 모든 DB 조작을 취소하고 원상복구 await connection.rollback(); console.error("정산 프로세스 중 오류 발생, 롤백되었습니다:", error); } finally { // 커넥션 반환 connection.release(); } }
