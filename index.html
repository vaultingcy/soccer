// 1. 경기 결과가 업데이트되었을 때 실행되는 정산 및 알림 함수 async function settleMatchAndNotify(matchId, finalHomeScore, finalAwayScore) { // DB에서 해당 경기를 예측한 유저들과 그들의 예측 스코어 가져오기 const predictions = await db.query("SELECT * FROM predictions WHERE match_id = ?", [matchId]); for (let pred of predictions) { let scoreEarned = 0; let resultMessage = ""; // 2. 점수 계산 로직 if (pred.home_score === finalHomeScore && pred.away_score === finalAwayScore) { scoreEarned = 3; // 스코어 정확히 맞춤 (대박) resultMessage = `[적중] 스코어(${finalHomeScore}:${finalAwayScore})를 정확히 맞추어 3점을 획득했습니다!`; } else if (Math.sign(pred.home_score - pred.away_score) === Math.sign(finalHomeScore - finalAwayScore)) { scoreEarned = 1; // 승무패만 맞춤 resultMessage = `[절반의 성공] 승패를 맞추어 1점을 획득했습니다. (결과 ${finalHomeScore}:${finalAwayScore})`; } else { scoreEarned = 0; // 틀림 resultMessage = `[아쉽습니다] 예측에 실패했습니다. (결과 ${finalHomeScore}:${finalAwayScore})`; } // 유저 점수 업데이트 await db.query("UPDATE users SET total_score = total_score + ? WHERE id = ?", [scoreEarned, pred.user_id]); // 3. 카카오톡 알림톡 발송 요청 (예시 템플릿 호출) await sendKakaoAlimtalk({ phoneNumber: pred.user_phone, templateCode: "MATCH_RESULT_ALERT", parameters: { "userName": pred.user_name, "matchName": pred.match_name, "result": resultMessage, "totalScore": pred.current_total + scoreEarned } }); } }
